def getSINR(h, w, noiseVar):
    # h: L x K x Nr x Nt  or n x Nr x Nt
    # w: Nt x Nl   (Note that Nl<= min(Nr,Nt))
    # Returns n x nl
    h = h.reshape(-1, h.shape[-2], h.shape[-1])          # n x Nr x Nt   (n=L*K if h is 4D)
    heff = np.matmul(h,w,axes=[(1,2),(0,1),(1,2)])       # n x Nr x Nl
    u, s, vH = np.linalg.svd(heff, full_matrices=False)  # n x Nr x Nl , n x Nl, n x Nl x Nl
    noisyInvS = 1/(np.square(s)+noiseVar)                # n x Nl
    # Calculating (V . noisyInvS . VH) is the same as:
    #  1) expanding dimensions of 'noisyInvS',  n x Nl =>                  n x Nl x 1
    #  2) calculating V.VH witch is the same square of magnitude of V      n x Nl x Nl
    #  3) doing elementwise multiplication (with boadcasting on last dim)  n x Nl x Nl
    #  4) summing on second axis                                           n x Nl
    gamma = 1/(noiseVar*(noisyInvS[:,:,None] * np.square(np.abs(vH))).sum(1)) - 1
    return gamma.real         # n x Nl

def getSINR2(h, w, noiseVar):
    # h: L x K x Nr x Nt  or n x Nr x Nt
    # w: Ncb x Nt x Nl   (Note that Nl<= min(Nr,Nt)) (Ncb: codebook size)
    # Returns Ncb x n x nl
    h = h.reshape(-1, h.shape[-2], h.shape[-1])          # n x Nr x Nt   (n=L*K if h is 4D)
    heff = np.matmul(h[None,:,:,:],w[:,None,:,:],axes=[(2,3),(2,3),(2,3)])       # Ncb x n x Nr x Nl
    u, s, vH = np.linalg.svd(heff, full_matrices=False)  # Ncb x n x Nr x Nl , Ncb x n x Nl, Ncb x n x Nl x Nl
    noisyInvS = 1/(np.square(s)+noiseVar)                # n x Nl
    # Calculating (V . noisyInvS . VH) is the same as:
    #  1) expanding dimensions of 'noisyInvS',  n x Nl =>                  Ncb x n x Nl x 1
    #  2) calculating V.VH witch is the same square of magnitude of V      Ncb x n x Nl x Nl
    #  3) doing elementwise multiplication (with boadcasting on last dim)  Ncb x n x Nl x Nl
    #  4) summing on second axis                                           Ncb x n x Nl
    gamma = 1/(noiseVar*(noisyInvS[:,:,:,None] * np.square(np.abs(vH))).sum(2)) - 1
    return gamma.real         # Ncb x n x Nl
